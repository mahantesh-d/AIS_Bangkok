--docker commit
	docker commit <container-id> docker-username/<repo-name>
	Example: docker commit 52e2ba965e93 mahanteshd4140/mongosedb
--Check the docker images
	docker images
--pick the newly created the images by commit
	docker tag <image-id> docker-username/<repo-name>:<add-the-name(basically docker name in hub)>
	docker tag d8aea20b924c mahanteshd4140/mongosedb:preinstallmongo
--push the docker images
	docker push docker-username/<repo-name>:<add-the-name(basically docker name in hub)>
	docker push mahanteshd4140/mongosedb:preinstallmongo
